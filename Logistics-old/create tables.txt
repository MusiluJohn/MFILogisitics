 ,Stock_ID INT NOT NULL
 ,Quantity_Ordered FLOAT NULL
 ,Stock_Ordercode VARCHAR(50) NULL
 ,PO_NO VARCHAR(50) NULL
 ,Sequence_NO INT NULL
 ,Quantity_Processed FLOAT NULL
 ,PRIMARY KEY CLUSTERED (Line_ID)
) ON [PRIMARY]
GO

GO
